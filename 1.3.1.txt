void harvestTheField(){
    repeat(2){
        repeat(4){
            moveForward();
            pickBeeper();
            turnRight();
            moveForward();
            turnLeft();
        }
        moveForward();
        moveForward();
        turnLeft();
        moveForward();
        moveForward();
        turnLeft();
        repeat(4){
            moveForward();
            pickBeeper();
            turnRight();
            moveForward();
            turnLeft();
        }
        turnAround();
    }
    turnAround();
}
