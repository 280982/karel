void straight(){
    dropBeeper();
    moveForward();
    if(frontIsClear() && !beeperAhead()){
        straight();
    }
    else{
        corner();
        turnRight();
    }
    pickBeeper();
    moveForward();
}

void corner(){
    turnLeft();
    if(!beeperAhead()){
        straight();
    }
    else{
        dropBeeper();
        moveForward();
        turnRight();
    }
}
